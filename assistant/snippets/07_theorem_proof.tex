\begin{theorem}[Pitágoras]
\label{thm:pitagoras}
Em um triângulo retângulo, o quadrado da hipotenusa é igual à soma dos quadrados dos catetos.
\end{theorem}

\begin{proof}
Seja $c$ a hipotenusa e $a, b$ os catetos. Então:
\begin{equation}
    c^2 = a^2 + b^2
\end{equation}
\end{proof}
